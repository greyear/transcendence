# Auth Service - Manual curl reference
# Matches current codebase: no username field, email-based auth, parameterless endpoints
# BASE_URL: https://localhost:8443/auth
# Replace <TOKEN_FROM_TEST_N> with the token value printed by that test
# Replace <GOOGLE_ID_TOKEN> with a fresh token from https://developers.google.com/oauthplayground
# Replace <USER_ID_FROM_TEST_22> with the id value returned by test 22


# ========== 1. NORMAL REGISTRATION TESTS ==========

# 1. Register new user - valid
curl -X POST 'https://localhost:8443/auth/register' \
  -H 'Content-Type: application/json' \
  -d '{"email":"normal1@test.local","password":"TestPass123!"}'

  201: {"email":"normal1@test.local","id":<id>,"token":"<token>","message":"Login successful"}


# 2. Register - duplicate email
curl -X POST 'https://localhost:8443/auth/register' \
  -H 'Content-Type: application/json' \
  -d '{"email":"normal1@test.local","password":"TestPass456@"}'

  409: {"error":"Resource exists"}


# 3. Register - invalid email format
curl -X POST 'https://localhost:8443/auth/register' \
  -H 'Content-Type: application/json' \
  -d '{"email":"notanemail","password":"TestPass123!"}'

  422: {"error":"Invalid email address"}


# 4. Register - password too short
curl -X POST 'https://localhost:8443/auth/register' \
  -H 'Content-Type: application/json' \
  -d '{"email":"user4@test.local","password":"Short1!"}'

  422: {"error":"The password doesn't match the password requirements"}


# 5. Register - password no uppercase
curl -X POST 'https://localhost:8443/auth/register' \
  -H 'Content-Type: application/json' \
  -d '{"email":"user5@test.local","password":"testpass123!"}'

  422: {"error":"The password doesn't match the password requirements"}


# 6. Register - missing email
curl -X POST 'https://localhost:8443/auth/register' \
  -H 'Content-Type: application/json' \
  -d '{"password":"TestPass123!"}'

  422: {"error":"Invalid email address"}


# ========== 2. NORMAL LOGIN TESTS ==========

# 7. Login - valid by email
curl -X POST 'https://localhost:8443/auth/login' \
  -H 'Content-Type: application/json' \
  -d '{"email":"normal1@test.local","password":"TestPass123!"}'

  200: {"token":"<token>","message":"Login successful"}


# 8. Login - wrong password
curl -X POST 'https://localhost:8443/auth/login' \
  -H 'Content-Type: application/json' \
  -d '{"email":"normal1@test.local","password":"WrongPass123!"}'

  401: {"error":"Password mismatch"}


# 9. Login - non-existent email
curl -X POST 'https://localhost:8443/auth/login' \
  -H 'Content-Type: application/json' \
  -d '{"email":"doesnotexist@test.local","password":"TestPass123!"}'

  404: {"error":"User not found"}


# 10. Login - missing email field
curl -X POST 'https://localhost:8443/auth/login' \
  -H 'Content-Type: application/json' \
  -d '{"password":"TestPass123!"}'

  404: {"error":"User not found"}


# 11. Login - extract token for later tests (same as test 7, copy the token value)
curl -X POST 'https://localhost:8443/auth/login' \
  -H 'Content-Type: application/json' \
  -d '{"email":"normal1@test.local","password":"TestPass123!"}'

  200: {"token":"<TOKEN_FROM_TEST_11>","message":"Login successful"}

# ========== 7. PASSWORD CHANGE TESTS ==========
# Requires USER_ID_FROM_TEST_22 and TOKEN_FROM_TEST_11

# 30. Change password - valid
curl -X PATCH 'https://localhost:8443/auth/change-password' \
  -H 'Content-Type: application/json' \
  -H 'Authorization: Bearer <TOKEN_FROM_TEST_11>' \
  -d '{"email":"normal1@test.local","userId":<USER_ID_FROM_TEST_22>,"password":"TestPass123!","newPassword":"NewPass456@"}'

  200: {"message":"Password updated successfully"}


# 31. Login - verify new password works
curl -X POST 'https://localhost:8443/auth/login' \
  -H 'Content-Type: application/json' \
  -d '{"email":"normal1@test.local","password":"NewPass456@"}'

  200: {"token":"<token>","message":"Login successful"}


# 32. Change password - no auth header
