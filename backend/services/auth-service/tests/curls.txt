# 1. Register new user - valid
curl -X POST 'http://localhost:3000/auth/register' \
  -H 'Content-Type: application/json' \
  -d '{"username":"normaluser1","email":"normal1@test.local","realname":"Normal User","password":"TestPass123!"}'

  201: {"username":"normaluser1","email":"normal1@test.local","realname":"Normal User"}


# 2. Register - duplicate username
curl -X POST 'http://localhost:3000/auth/register' \
  -H 'Content-Type: application/json' \
  -d '{"username":"normaluser1","email":"other@test.local","realname":"Other","password":"TestPass123!"}'

  409: {"error":"Resource exists"}

# 20. Change password - valid (use token from step 11)
curl -X PATCH 'http://localhost:3000/auth/change-password/normaluser1' \
  -H 'Content-Type: application/json' \
  -H 'Authorization: Bearer <TOKEN_FROM_STEP_11>' \
  -d '{"password":"TestPass123!","newPassword":"NewPass456@"}'

  200: {"message":"Password updated successfully"}
